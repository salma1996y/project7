tr -s [:space:][:punct:] \\n < file.txt | sed 's/[^a-z  A-Z]//g' | tr [:upper:] [:lower:] | sort | uniq | wc -l


استبدال علامات الترقيم بمسافه وضغط الكلمات ووضعها بسطر جديد = tr -s [:space:][:punct:] \\n
يعزل الكلمات بالحروف ألكبتيل والسمول =  sed 's/[^a-z  A-Z]//g'
يحول الحروف من كابيل الى سمول = tr [:upper:] [:lower:]
الترتيب الأبجدي = sort
uniq  = إيجاد الكلمات الفريدة
عد الكلمات = wc -w
توجيه المخرجات لامر اخر = |
